.ORIG x10
.START x10

MOV w X200 R1
MOV w xA R2
CALL FIBO
HALT

//FUNCTION WHICH GENERATES FIBONACCI NOS
//AND STORES AT MEMORY SPECIFIED IN R1
//HOW MANY TO GENERATE - R2
FIBO: MOV b x1 R3
    MOV b x0 R4
LOOP: COMP b x0 R2
    JEQ DONE
    MOV w R3 R5
    ADD w R4 R5
    STORE w [R1] R5
    ADD w x4 R1
    MOV w R3 R4
    MOV w R5 R3
    SUB w x1 R2
    JMP LOOP
DONE: RET
