.ORIG x10
.START x10
MOV b x5 R1

LOOP: IN x0 R2
    COMP b R1 R2
    JNE LOOP
HALT





.ORIG x10
.START x10

//R1 R2 R3 ARE USED TO STORE INPUTS FROM
//PORTS 1 2 AND 3
//PORT 4 IS STATUS REGISTER
//OUTPUT IS AT PORT 5

LOOP: IN x4 R4
    COMP b x11 R4
    JNE LOOP

//NOW INPUT IS READY
IN x1 R1
IN x2 R2
IN x3 R3
ADD w R1 R3
ADD w R2 R3
CALL DIV
OUT x5 R6

HALT

//DIVIDE VAL IN R3 BY 3 AND STORE ANS IN R6

DIV: MOV w R3 R7
    MOV b x0 R6
 DIVLOOP: COMP w x3 R7
    JGT DONE
    ADD w x1 R6
    SUB w x3 R7
    JMP DIVLOOP
DONE: RET

.END
